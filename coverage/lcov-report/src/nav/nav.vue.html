<template>
    <div class="lf-nav" :class="{vertical}">
        <slot></slot>
    </div>
</template>
 
<script>
    export default {
        name: "LiFaNav",
        props: {
            selected: {
                type: String,
            },
            vertical: {
                type: Boolean
            }
        },
        data(){
            return {
                item: [],
                namePath: []
            }
        },
        provide(){
            return {
                root: this,
                vertical: this.vertical
            }
        },
        mounted() {
            this.updateChildren()
            this.listenToChildren()
        },
        updated() {
            this.updateChildren()
            this.listenToChildren()
        },
        methods: {
            addItem(vm){
                this.item.push(vm)
            },
            updateChildren: function () {
                this.item.forEach(vm => {
                    if(this.selected === vm.name) {
                        vm.selected = true
                    } else {
                        vm.selected = false
                    }
                })
            },
            listenToChildren(){
                this.item.forEach(vm=>{
                    vm.$on('update:selected', (name) => {
                        this.$emit('update:selected',name)
                    })
                })
 
            }
        }
 
    }
</script>
