<template>
    <div class="lf-nav-item" :class="{active: selected,vertical}" @click="onClick"
    :data-name="name"
    >
        <slot></slot>
    </div>
</template>
 
<script>
    export default {
        name: "LiFaNavItem",
        inject: ['root','vertical'],
        props: {
            name: {
                type: String,
                required: true
            }
        },
        data(){
            return {
                selected: undefined
            }
        },
        created(){
          this.root.addItem(this)
        },
        methods: {
            onClick(){
                this.root.namePath = []
                this.$parent.updateNamePath && this.$parent.updateNamePath()
                this.$emit('update:selected',this.name)
            }
        }
    }
</script>
